# 获取从前数(从后数)，size之外的元素

/**
 * 
 * @param {Array} arr 
 * @param {Number} size 
 */
function drop(arr, size){
  return arr.filter((value, index) => index >= size);
}

function dropRight(arr, size){
  return arr.filter((value, index) => index < arr.length - size);
}
