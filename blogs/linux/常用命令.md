- 查看当前进程
  ps -f
- 查看进程
  ps -ef | grep nginx

- 主进程 id 存储位置（master）子进程（worker）
  nginx/logs/nginx.pid 只有主进程 id

- 查看本机支持的 shell 解析器
  cat /etc/shells

- 查看本机默认 shell
  echo $SHELL

- 创建文件夹
  mkdir

- 创建文件
  touch

- 给文件加可执行权限
  chmod +x 文件

- 查看所有系统全局变量
  env

- 查看$USER
  printenv USER

- 查看定义的所有变量（系统和用户）
  set | less // | less 翻页查看

- 系统 bin 目录
  $PATH
