##  原生方法

```JavaScript
let arr = [1,2,5,4,2,1]
let arr2 = []
for (let i=0; i<arr.length; i++) {
	let item = arr[i]
	if (arr2.indexOf(item) === -1) {
		arr2.push(item)
	}
}
```

##  ES6
```JavaScript
function uniqArr(arr) {
  let set = new Set(arr)
  let arr2 = []
  for (let i of set) {
    arr2.push(i)
  }
  return arr2
}
// 简写
let uniqArr = arr => [...new Set(arr)]
```

##  reduce方法
```JavaScript
let result = arr.sort.reduce((acc, cur) => {
  if (acc.length === 0 || acc[length-1] !== cur) {
    acc.push(cur)
  }
  return acc
},[])


let result = arr.reduce((acc, cur) => {
 !acc.includes(cur) && acc.push(cur)  
 return acc
},[])
```
