##  输入 20200820164516 返回 2020年08月20日 16:45:16
```js
function fn(num) {
    let str = num + ''
    return `${str.substring(0, 4)}年${str.substring(4, 6)}月${str.substring(6, 8)}日 ${str.substring(8, 10)}:${str.substring(10, 12)}:${str.substring(12, 14)}`
}
console.log(fn(20200820164516))
```

##  字符串大小写转换
```js
let reg = /[a-z]/
let REG = /[A-Z]/
function fn(str) {
    let res = ''
    for (let i of str) {
        if (i.match(reg)) {
            res += i.toUpperCase()
        } else if (i.match(REG)) {
            res += i.toLowerCase()
        }
    }
    return str
}
console.log(fn('aBc'))
```
