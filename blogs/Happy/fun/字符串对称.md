```js
function str(str) {
    let newStr = str.split("").reverse().join("")
    if(newStr === str) {
        return true
    }else {
        return false
    }
}

function str(str) {
    for(let i = 0; i < parseInt(str.length / 2); i++) {
        if(str.charAt(i) !== str.charAt(str.length - i - 1)) {
            return false
        }
    }
    return true
}
```
