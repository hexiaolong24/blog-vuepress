// 直接反转单词
let s = 'www.baidu.com'
function fn(s) {
  let arr = s.split('.')
  let l = arr.length
  let res = ''
  while(--l >= 0) {
    res += arr[l] + (l === 0 ? '' : '.')
  }
  return res
}

function fn(s) {
  let arr = s.split('.')
  let r = arr.length
  let l = -1
  while(++l < r--) {
    [arr[l], arr[r]] = [arr[r], arr[l]]
  }
  return arr.join('.')
}
console.log(fn(s))


// 先反转整体，再反转单词
let s = 'www.baidu.com'
function fn(s) {
  let arr = Array.from(s)
  reverse(arr, 0, arr.length-1)
  let l = 0
  for(let i = 0; i< arr.length; i++) {
    if(arr[i] === '.' || i === arr.length -1) {
      reverse(arr, l, i-1)
      l = i+1
    }
  }
  return arr.join('')
}
function reverse(s, l, r) {
  while(l < r) {
    [s[l], s[r]] = [s[r], s[l]]
    l++
    r--
  }
}
console.log(fn(s))
