```js
// 一
let arr = ["绿", "绿", "绿", "红", "红", "黄"];
let num = 0;
setInterval(() => {
  console.log(arr[num]);
  num++;
  if (num > 5) {
    num = 0;
  }
}, 1000);

// 二
function fn() {
  console.log("green");
  setTimeout(() => {
    console.log("yellow");
    setTimeout(() => {
      console.log("red");
      setTimeout(fn, 2000);
    }, 1000);
  }, 3000);
}
fn();

// 三 红绿灯输出
function helper(time) {
  return new Promise((resolve) => {
    setTimeout(resolve, time);
  });
}
(async () => {
  while (1) {
    console.log("green");
    await helper(5000);
    console.log("yellow");
    await helper(2000);
    console.log("red");
    await helper(3000);
  }
})();
```
