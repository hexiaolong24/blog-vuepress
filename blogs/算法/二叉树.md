##  TIPS
-   二叉树题目的要求细化，搞清楚根节点应该做什么，然后剩下的事情抛给前/中/后序的遍历框架就行了，我们千万不要跳进递归的细节里，你的脑袋才能压几个栈呀。

##  深度
二叉树节点的深度：指从根节点到该节点的最长简单路径边的条数。

##  高度
二叉树节点的高度：指从该节点到叶子节点的最长简单路径边的条数。

##  二叉树前序遍历 中左右
root root.left root.right
##  二叉树中序遍历 左中右
root.left root root.right
##  二叉树后序遍历 左右中
root.left root.right root

```js
var inorderTraversal = function(root) {
    const res = []
    if(root == null) {
        return res
    }
    const stack = []
    let cur = root
    while(stack.length !== 0 || cur) {
      if(cur) {
        stack.push(cur)
        cur = cur.left
      }else {
        cur = stack.pop()
        res.push(cur.val)
        cur = cur.right
      }
    }
    return res
```

##  543 二叉树的最大直径
```js
var diameterOfBinaryTree = function (root) {
  let maxDiameter = 0;
  const dfs = (root) => {
    // 访问到空节点了，返回0
    if (root == null) return 0;
    // 左子树为根的子树的的深度
    let leftMax = dfs(root.left);
    let rightMax = dfs(root.right);
    maxDiameter = Math.max(maxDiameter, leftMax + rightMax);
    // 返回该节点为根的子树的的深度
    return Math.max(leftMax, rightMax) + 1;
  };
  dfs(root);
  return maxDiameter;
};
```

##  101 对称二叉树
```js
var isSymmetric = function(root) {
    // 如果只有根节点 是
    if(!root) return true
    let order = (left, right) => {
        // 1. 有左没右 或者 有右没左
        if((left && !right) || (right && !left)) {
            return false
        }else if(!left && !right) {
            // 2. 直到最后都没有子节点就返回对， 如果判断其中一个就返回对是不对的， 因为要一直递归下去
            return true
        }else if(left.val !== right.val) {
            // 如果传入的两个值不相等为false
            return false
        }
        // 1 2 3 4 
        // 1 4
        let outside = order(left.left, right.right)
        // 2 3
        let inside = order(left.right, right.left)
        return outside && inside
    }
    return order(root.left, root.right)
};
```

```js
var countNodes = function(root) {
    if(root === null) return 0
    let left = root.left
    let right = root.right
    let leftHeight = rightHeight = 0
    while(left) {
        leftHeight++
        left = left.left
    }
    while(right) {
        rightHeight++
        right = right.right
    }
    if(leftHeight === rightHeight) {
```

```js
var isBalanced = function(root) {
    let helper = (root) => {
        if(!root) {
            return 0
        }
        let leftDep = helper(root.left)
        if(leftDep === -1) {
            return -1
        }
```

##  236 二叉树的最近公共祖先
```js
var lowestCommonAncestor = function(root, p, q) {
    const helper = (root, p, q) => {
        // 确定终止条件
        if(root === null || root === p || root === q) {
            return root
        }
        // 单层递归逻辑(后续遍历，先递归，后处理逻辑)
        let left = helper(root.left, p, q)
        let right = helper(root.right, p, q)
        // 找到谁返回谁
        if(left && !right) {
            return left
        }else if(!left && right) {
            return right
        }else if(left && right){ // 都找到 祖先就是root节点
            return root
        }else { // 都没有返回null
            return null
        }
    }
    return helper(root, p, q)
};
```

##  450 删除搜索二叉树中的节点
```js
var deleteNode = function(root, key) {
    // 没找到
    if(!root) return null
    // 找打了
    if(root.val === key) {
        // 没有子节点，直接返回null
        if(!root.left && !root.right) {
            return null
        // 有右 没有左，则有的直接补位
        }else if(!root.left && root.right) {
            return root.right
        // 同上
        }else if(!root.right && root.left) {
            return root.left
        }else {
            // 左右都有，先找到右子节点的最左子节点
            let cur = root.right
            while(cur.left) {
                cur = cur.left
            }
            // 将root左子节点放到找到的最左子节点
            cur.left = root.left
            // 返回root.right,相当于删除root
            root = root.right
            return root
        }
    }
    if(key > root.val) {
        root.right = deleteNode(root.right, key)
    } 
    if(key < root.val) {
        root.left = deleteNode(root.left, key)
    }
    return root
};
```

##  538 把二叉搜索数转换为累加树

```js
var convertBST = function(root) {
    let pre = 0
    const helper = (cur) => {
        if(!cur) return
        helper(cur.right)
        cur.val += pre
        pre = cur.val
        helper(cur.left)
    }
    helper(root)
    return root
};
```

##  114 二叉树展开为链表
```js
var flatten = function(root) {
    if (root == null) return;

    flatten(root.left);
    flatten(root.right);

    let left = root.left;
    let right = root.right;

    root.left = null;
    root.right = left;

    let p = root;
    while (p.right != null) {
        p = p.right;
    }
    p.right = right;
};
```

##  669 修剪二叉搜索树
```js
var trimBST = function(root, low, high) {
    if(!root) return null
    if(root.val > high) {
        root.left = trimBST(root.left, low, high)
        return root.left
    }
    if(root.val < low) {
        root.right = trimBST(root.right, low, high)
        return root.right
    }
    root.left = trimBST(root.left, low, high)
    root.right = trimBST(root.right, low, high)
    return root
};
```

##  652 寻找重复的子树
```js

var findDuplicateSubtrees = function(root) {
    const count = new Map()
    const ans = []

    const collect = (node) => {
        if (!node) {
            return '#'
        }
        const key = node.val + ',' + collect(node.left) + collect(node.right)
        if (count.has(key)) {
            count.set(key, count.get(key) + 1)
        } else {
            count.set(key, 1)
        }
        if (count.get(key) === 2) {
            ans.push(node)
        }
        return key
    }

    collect(root)
    return ans
};
```
