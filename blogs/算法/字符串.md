##  344. 反转字符串
- 双指针
```js
var reverseString = function(s) {
  let l = -1, r = s.length
  while(++l < --r) {
    [s[l], s[r]] = [s[r], s[l]]
  }
  return s
};
```

##  541. 字符串反转 II
```js
var reverseStr = function(s, k) {
  // 字符串转数组
  let arr = s.split('')
  let len = s.length 
  for(let i = 0; i < len; i += 2*k) {// 每次间隔2k
    let l = i -1 // 因为 0 1长度为2      -1 + 2 = 1（0 1）
    // 如果超过 k ,只反转k 如果没超过k 则反转剩余所有的
    let r = i + k > len ? len : i + k
    while(++l < --r) {
      [arr[l], arr[r]] = [arr[r], arr[l]]
    }
  }
  return arr.join("")
};
```

##  剑指Offer05. 替换空格
```js
var replaceSpace = function(s) {
    let newStr = Array.from(s)
    let count = 0
    for(let i = 0; i < newStr.length; i++) {
        if(newStr[i] === ' ') {
            count++
        }
    }
    let left = newStr.length - 1,
        right = left + 2*count
    while(left >= 0) {
        if(newStr[left] === ' ') {
            newStr[right--] = '0'
            newStr[right--] = '2'
            newStr[right--] = '%'
            left--
        }else {
            newStr[right--] = newStr[left--]
        }
    }
    return newStr.join("")
};
```
